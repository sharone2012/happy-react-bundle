## 🚀 UI IMPLEMENTATION GUIDE

### **React component structure:**

```javascript
// State
const [selectedSoil, setSelectedSoil] = useState('Inceptisol');
const [selectedObjective, setSelectedObjective] = useState('value_for_money');
const [organisms, setOrganisms] = useState([]);

// Objective button click handler
const handleObjectiveClick = async (objective) => {
  setSelectedObjective(objective);
  
  const { data } = await supabase.rpc('rank_organisms_by_objective', {
    p_soil_type: selectedSoil,
    p_objective: objective,
    p_limit: 10
  });
  
  setOrganisms(data);
};

// Render buttons
const objectives = [
  { id: 'value_for_money', icon: '💰', label: 'Best Value' },
  { id: 'p_release', icon: '🧪', label: 'P-Release' },
  { id: 'lignin', icon: '🌾', label: 'Lignin' },
  { id: 'n_fixing', icon: '🌱', label: 'N-Fixing' },
  { id: 'bsf_ready', icon: '🐛', label: 'BSF-Ready' },
  { id: 'n_trap', icon: '🪤', label: 'N-Trap' },
  { id: 'k_leaching', icon: '⚡', label: 'K-Retain' },
  { id: 'liming_response', icon: '🧂', label: 'Post-Lime' },
  { id: 'iron_management', icon: '🧲', label: 'Fe-Mgmt' },
  { id: 'acid_tolerance', icon: '🔋', label: 'Acid Tol' },
  { id: 'n2o_safe', icon: '✅', label: 'N2O-Safe' },
  { id: 'soil_specific', icon: '🎯', label: 'Optimized' }
];
```

### **Priority highlighting per soil:**

```javascript
// Conditional styling based on soil + objective priority
const getPriority = (soil, objective) => {
  const priorities = {
    'Inceptisol': ['value_for_money', 'lignin', 'bsf_ready'],
    'Ultisol': ['p_release', 'value_for_money', 'n_trap'],
    'Oxisol': ['p_release', 'liming_response', 'acid_tolerance'],
    'Histosol': ['n2o_safe', 'acid_tolerance', 'value_for_money'],
    'Spodosol': ['n_trap', 'k_leaching', 'p_release'],
    'Andisol': ['p_release', 'liming_response', 'acid_tolerance']
  };
  
  return priorities[soil]?.includes(objective) ? 'HIGH' : 'NORMAL';
};

// Button styling
<button
  className={`objective-btn ${getPriority(selectedSoil, obj.id) === 'HIGH' ? 'priority-high' : ''}`}
  onClick={() => handleObjectiveClick(obj.id)}
>
  {obj.icon} {obj.label}
</button>
```
